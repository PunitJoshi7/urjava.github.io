                import javax.swing.JButton;
import javax.swing.JFrame;
import javax.swing.JLabel;
import javax.swing.JPasswordField;
import javax.swing.JTextArea;
import javax.swing.JScrollPane;
import javax.swing.JPanel;
import java.awt.BorderLayout;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;

public class TextAreaPasswordFieldExample extends JFrame {

    private JTextArea textArea;
    private JPasswordField passwordField;

    public TextAreaPasswordFieldExample() {
        // Set the title of the frame
        setTitle("Text Area and Password Field Example");

        // Create a panel to hold components
        JPanel panel = new JPanel();

        // Create a text area
        textArea = new JTextArea(5, 20);
        JScrollPane textAreaScrollPane = new JScrollPane(textArea);
        panel.add(textAreaScrollPane);

        // Create a password field
        JLabel passwordLabel = new JLabel("Password:");
        passwordField = new JPasswordField(20);
        panel.add(passwordLabel);
        panel.add(passwordField);

        // Create a button to display the entered values
        JButton displayButton = new JButton("Display Values");
        displayButton.addActionListener(new ActionListener() {
            @Override
            public void actionPerformed(ActionEvent e) {
                displayValues();
            }
        });
        panel.add(displayButton);

        // Add the panel to the frame
        add(panel, BorderLayout.CENTER);

        // Set the size of the frame
        setSize(400, 200);

        // Set the default close operation
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);

        // Set the frame to be visible
        setVisible(true);
    }

    private void displayValues() {
        String textAreaValue = textArea.getText();
        char[] passwordFieldValue = passwordField.getPassword();

        System.out.println("Text Area Value: " + textAreaValue);
        System.out.println("Password Field Value: " + new String(passwordFieldValue));
    }

    public static void main(String[] args) {
        // Create an instance of the TextAreaPasswordFieldExample class
        TextAreaPasswordFieldExample frame = new TextAreaPasswordFieldExample();
    }
}

        option2Radio.setBounds(120, 40, 100, 20);
        option3Radio.setBounds(120, 70, 100, 20);

        submitButton.setBounds(10, 110, 100, 30);
        resultLabel.setBounds(120, 110, 200, 30);

        // Add action listener to the submit button
        submitButton.addActionListener(new ActionListener() {
            @Override
            public void actionPerformed(ActionEvent e) {
                updateResultLabel();
            }

        panel.add(javaCheckbox);
        panel.add(pythonCheckbox);
        panel.add(cppCheckbox);
        panel.add(option1Radio);
        panel.add(option2Radio);
        panel.add(option3Radio);
        panel.add(submitButton);

        setVisible(true);

        if (pythonCheckbox.isSelected()) {
            selectedOptions.append(" Python");

        if (cppCheckbox.isSelected()) {
            selectedOptions.append(" C++");

        if (option1Radio.isSelected()) {
            selectedOptions.append(" Option 1");
        } else if (option2Radio.isSelected()) {
            selectedOptions.append(" Option 2");
        } else if (option3Radio.isSelected()) {
            selectedOptions.append(" Option 3");
        }

        // Update the label with the selected options
        JLabel resultLabel = (JLabel) getContentPane().getComponent(0).getComponent(6);
        resultLabel.setText(selectedOptions.toString());
    }

                import javax.swing.JButton;
import javax.swing.JFrame;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;

public class FrameSwitchingExample {

    private JFrame frame1, frame2;

    public FrameSwitchingExample() {
        // Create the first frame
        frame1 = new JFrame("Frame 1");
        frame1.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);

        // Create a button on the first frame
        JButton switchButton = new JButton("Switch to Frame 2");
        switchButton.setBounds(50, 50, 150, 30);

        // Add an ActionListener to the button
        switchButton.addActionListener(new ActionListener() {
            @Override
            public void actionPerformed(ActionEvent e) {
                // Switch to the second frame when the button is clicked
                switchToFrame2();
            }
        });

        // Add the button to the first frame
        frame1.add(switchButton);

        // Set the size of the first frame
        frame1.setSize(250, 150);

        // Set the frame to be visible
        frame1.setLayout(null);
        frame1.setVisible(true);
    }

    private void switchToFrame2() {
        // Create the second frame
        frame2 = new JFrame("Frame 2");
        frame2.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);

        // Create a button on the second frame
        JButton backButton = new JButton("Back to Frame 1");
        backButton.setBounds(50, 50, 150, 30);

        // Add an ActionListener to the button
        backButton.addActionListener(new ActionListener() {
            @Override
            public void actionPerformed(ActionEvent e) {
                // Switch back to the first frame when the button is clicked
                switchToFrame1();
            }
        });

        // Add the button to the second frame
        frame2.add(backButton);

        // Set the size of the second frame
        frame2.setSize(250, 150);

        // Set the frame to be visible
        frame2.setLayout(null);
        frame1.setVisible(false);
        frame2.setVisible(true);
    }

    private void switchToFrame1() {
        // Dispose of the second frame and show the first frame
        frame2.dispose();
        frame1.setVisible(true);
    }

    public static void main(String[] args) {
        // Create an instance of the FrameSwitchingExample class
        FrameSwitchingExample example = new FrameSwitchingExample();
    }
}
